ADT String{
  数据对象：
  数据关系：
  基本操作：
   StrAssign (&T, chars) //生成一个其值等于chars的串T
   
   DestroyString(&S)//销毁串S
   
   StrCopy (&T, S)//由串S复制得串T
   
   StrLength(S)//返回S的元素个数，串的长度
   
   StrCompare (S, T)//若S>=T,返回>0,若S<T,返回<0
   
   StrEmpty (S)//若串为空，返回TRUE，否则返回FALSE
   
   SubString (&Sub, S, pos, len)用Sub返回串S的第pos个字符起长度为len的字串
   
   ClearString (&S) 清空S为空串
   
   Index (S, T, pos)//// T为非空串。若主串S中第pos个字符之后存在与T相等的子串，
                    // 则返回第一个这样的子串在S中的位置，否则返回0
   Replace (&S, T, V)
   
   StrInsert (&S, pos, T)
   
   StrDelete (&S, pos, len)

   Concat(SString &T, SString S1, SString S2) // 用T返回由S1和S2联接而成的新串。若未截断，则返回TRUE，否则FALSE
} ADT String
